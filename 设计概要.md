# 投资项目管理（MongoDB数据结构）

## 项目 (project)
```javascript
{
  _id: ObjectId,
  项目名称: String,
  项目代号: String,
  成本价: Number,
  当前价: Number,
  股数: Number,
  仓位: Number,
  成本金额: Number,
  当前金额: Number,
  盈亏金额: Number,
  项目盈亏率: Number,
  自主盈亏率: Number,
  创建时间: Date,
  完成时间: Date
}
```

## 交易 (transaction)
```javascript
{
  _id: ObjectId,
  项目ID: ObjectId,
  项目名称: String,
  状态: String, // 计划 / 完成
  交易名称: String,
  交易类型: String,
  警告方向: String,
  距离: Number,
  交易价: Number,
  止盈价: Number,
  止损价: Number,
  股数: Number,
  仓位: Number,
  交易金额: Number,
  最大亏损额: Number,
  最大亏损率: Number,
  最大盈利额: Number,
  最大盈利率: Number,
  创建时间: Date,
  交易时间: Date
}
```

## 基金 (fund)
```javascript
{
  _id: ObjectId,
  基金名称: String,
  成本金额: Number,
  当前金额: Number,
  累计净值: [
    {
      时间: Date,
      累计净值: Number
    }
  ],
  交易列表: [
    {
      _id: ObjectId,
      交易类型: String,
      交易净值: Number,
      交易金额: Number,
      交易时间: Date
    }
  ]
}
```

## 统计 (statistics)
```javascript
{
  _id: ObjectId,
  年月: String,
  类型: String,
  自主盈亏金额: Number,
  自主盈亏率: Number,
  基金盈亏金额: Number,
  基金盈亏率: Number,
  总盈亏金额: Number,
  总盈亏率: Number
}
```

## 总览 (overview)
```javascript
{
  _id: ObjectId,
  
  // 项目投资概况（自主投资）
  自主投资金额: Number,
  自主持仓金额: Number,
  自主盈亏金额: Number,
  自主盈亏率: Number,
  自主仓位: Number,
  
  // 基金投资概况（可选）
  基金投资金额: Number,
  基金盈亏金额: Number,
  基金盈亏率: Number,
  
  // 整体投资情况
  总投资金额: Number,
  总盈亏金额: Number,
  总盈亏率: Number,
  
  // 项目列表引用
  项目: [ObjectId],

  // 基金引用
  基金: [ObjectId],

  月度统计列表: [ObjectId],
  年度统计列表: [ObjectId]
}
```
